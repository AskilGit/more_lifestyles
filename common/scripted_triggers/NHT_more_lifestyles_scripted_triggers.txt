nht_fortified_barony_is_valid_trigger = {
	title_province = {
		NOT = { has_province_modifier = nht_reinforced_barony_modifier }
	}
	trigger_if = {
		limit = {
			holder = $CHARACTER$
		}
	}
	trigger_else = {
		holder.primary_title.tier = tier_barony
	}
}
